⚙ How this uses Temper

Entity: Signal

Each signal is a Temper entity: Sent → Processing → Done. Agents POST to create, receivers fire actions to transition.

# signal.ioa.toml
[entity]
name = "Signal"
initial_status = "Sent"

[vars]
has_response = { type = "bool", init = false }

[[actions]]
name = "Acknowledge"
from = ["Sent"]
to = "Processing"

[[actions]]
name = "Respond"
from = ["Processing"]
to = "Done"
effects = [{ set = { var = "has_response", value = true } }]

[[actions]]
name = "Dismiss"
from = ["Sent", "Processing"]
to = "Done"

API Flow

# Haku sends a signal to Calcifer
POST /tdata/Signals
{ "entity_id": "s-001",
  "fields": { "from": "haku", "to": "calcifer",
              "body": "PROP-028 shipped" } }

# Calcifer acknowledges → status becomes Processing
POST /tdata/Signals('s-001')/Temper.Acknowledge

# Can't Acknowledge a Done signal → 409 Conflict
